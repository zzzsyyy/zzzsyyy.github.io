  • 判断奇偶数

    //一般般
func EvenOrOdd(number int) string {
  if number % 2 == 0 {
      return "Even"
    }
    return "Odd"
}

// 统一变量，便于操作
func EvenOrOdd(number int) string {
  str := "Odd"
  if (number % 2 == 0) {
    str = "Even"
  }
  return str
}

//用num&1判断奇偶
func EvenOrOdd(number int) string {
  return []string{"Even", "Odd"}[number & 1]
}
